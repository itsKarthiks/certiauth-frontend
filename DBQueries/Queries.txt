# 1
-- Create Users Table
CREATE TABLE users (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  password TEXT NOT NULL,
  role TEXT DEFAULT 'student' CHECK (role IN ('admin', 'student')),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);

-- Create Certificates Table
CREATE TABLE certificates (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  certificate_id TEXT UNIQUE NOT NULL, 
  student_name TEXT NOT NULL,
  student_id TEXT NOT NULL,
  program TEXT NOT NULL,
  cgpa TEXT NOT NULL,
  issue_date DATE DEFAULT CURRENT_DATE,
  digital_signature TEXT NOT NULL,
  is_revoked BOOLEAN DEFAULT FALSE,
  revocation_reason TEXT DEFAULT '',
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);

-- 1. Create the function that does the actual deleting
create or replace function public.handle_deleted_profile()
returns trigger
language plpgsql
security definer -- This allows the script to bypass normal security to reach the hidden auth table
as $$
begin
  -- Delete the user from the hidden Auth system where the IDs match
  delete from auth.users where id = old.id;
  return old;
end;
$$;

-- 2. Create the trigger that watches the profiles table
drop trigger if exists on_profile_deleted on public.profiles;
create trigger on_profile_deleted
  after delete on public.profiles
  for each row execute procedure public.handle_deleted_profile();
